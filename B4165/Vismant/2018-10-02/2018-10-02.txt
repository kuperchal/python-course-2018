from math import sqrt 
def discriminant(first_coeff, second_coeff, third_coeff):
    return sqrt((second_coeff**2) - (4*first_coeff*third_coeff))
 
 
    
    
print("Вычисляем дискриминант квадратного уравнения")
a = float(input("Введите первый коэффициент:  ")) 
b = float(input("Введите второй коэффициент:  ")) 
c = float(input("Введите третий коэффициент:  "))

    
if discriminant(a,b,c) == 0:
       first_root = -b/2*a  
       print(first_root) 
elif discriminant(a,b,c) > 0:
       discriminant1 = discriminant(a,b,c) 
       first_root = -b + discriminant1/2*a 
       second_root = -b - discriminant1/2*a 
       print("Первый корень:", first_root, "Второй корень:", second_root)
else:  print ("Корней нет")
