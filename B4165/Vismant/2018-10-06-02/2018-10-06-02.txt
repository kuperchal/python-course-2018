class Cell:
  def __init__(self,n):
    self.numb = n
  def out(self):
    print (self.numb)

class A(Cell):
 def multi(self,m):
   self.numb *= m

class B(Cell):
  def inlist(self):
    self.inlist = list(str(self.numb))
  def out(self):
    i = 0
    while i < len(self.inlist):
      print (self.inlist[i])
      i += 1

obj1 = A(15)
obj2 = B('acrostih')

obj1.multi(3)
